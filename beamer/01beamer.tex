\begin{frame}{Introduction to the problem}
   Solving a \textbf{\textcolor{BrickRed}{PDE}} means to find a function $u$ such that

   \begin{equation*}
      \mathcal{L}u=f
   \end{equation*}

   where $\mathcal{L}$ is a \underline{differential operator} and $f$ is a \underline{source term}.

   \vspace{0.3cm}

   The equation holds in a domain $\Omega$ and is completed by prescribing \textbf{boundary conditions} on $\partial\Omega$.

   \vfill

   \begin{figure}[H]
      \centering
      \begin{tikzpicture}

         \node[align=center,text width=3.5cm] at (0,0) (A) {In most physical applications $\mathcal{L}$ is a \underline{second-order} operator};

         \node at (5,0) (B) {\footnotesize{\textbf{Heat} equation: $\mathcal{L}=\frac{\partial}{\partial t}-\Delta$}};

         \node at (4.85,1) (C) {\footnotesize{\textbf{Poisson} equation: $\mathcal{L}=\Delta$}};

         \node at (5.25,-1) (D) {\footnotesize{\textbf{Wave} equation: $\mathcal{L}=\frac{\partial^2}{\partial t^2}-c^2\Delta$}};

         \draw[->] (A) -- (B);
         \draw[->] (2.25,0) -- (2.25,1) -- (C);
         \draw[->] (2.25,0) -- (2.25,-1) -- (D); 
      \end{tikzpicture}
   \end{figure}
\end{frame}
